rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Default: Authenticated users can read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Public read-only collections
    match /public_content/{itemId} {
      allow read: if true;
      allow write: if false;
    }
    
    // SaaS specific data
    match /tenants/{tenantId} {
      allow read: if request.auth != null && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.tenantId == tenantId;
      allow write: if false; // Admin only or managed via Functions
    }
  }
}
